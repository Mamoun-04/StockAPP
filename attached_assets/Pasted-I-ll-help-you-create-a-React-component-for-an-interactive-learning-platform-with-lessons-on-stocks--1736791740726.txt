I'll help you create a React component for an interactive learning platform with lessons on stocks. I'll include nice formatting and UI elements that would work well for a trading education platform.



```tsx
import React, { useState } from 'react';
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { ScrollText, BookOpen, MessageCircle, GraduationCap, ChevronRight } from 'lucide-react';

const TradingLessons = () => {
  const [selectedDifficulty, setSelectedDifficulty] = useState('all');
  const [selectedTopic, setSelectedTopic] = useState('all');

  const difficulties = ['Beginner', 'Intermediate', 'Advanced'];
  const topics = ['Stock Basics', 'Technical Analysis', 'Fundamental Analysis', 'Risk Management'];

  const lessons = [
    {
      title: "Introduction to Stock Markets",
      description: "Learn the basics of how stock markets work, what stocks are, and how trading happens.",
      difficulty: "Beginner",
      topic: "Stock Basics",
      duration: "15 mins",
      content: `
        The stock market is a crucial part of the global economy where shares of publicly traded companies are bought and sold. When you buy a stock, you're purchasing a small ownership stake in a company.

        Key concepts covered in this lesson:
        • What is a stock and how does it represent ownership
        • How stock markets function as a marketplace
        • Different types of stock exchanges
        • Basic terminology every trader should know

        The primary way to make money in stocks is through:
        1. Capital appreciation - when the stock price increases
        2. Dividends - regular payments some companies make to shareholders
      `
    },
    {
      title: "Reading Candlestick Patterns",
      description: "Master the art of interpreting candlestick charts and common patterns.",
      difficulty: "Intermediate",
      topic: "Technical Analysis",
      duration: "25 mins",
      content: `
        Candlestick charts are a fundamental tool in technical analysis, providing visual insights into price movements and market psychology.

        This lesson covers:
        • Basic structure of a candlestick
        • Common candlestick patterns
        • How to interpret different formations
        • Using patterns for trading decisions

        We'll explore key patterns like:
        • Doji
        • Hammer and Hanging Man
        • Engulfing patterns
        • Morning and Evening stars
      `
    },
    {
      title: "Advanced Risk Management Strategies",
      description: "Learn sophisticated approaches to managing trading risk and portfolio exposure.",
      difficulty: "Advanced",
      topic: "Risk Management",
      duration: "30 mins",
      content: `
        Effective risk management is crucial for long-term trading success. This advanced lesson explores sophisticated strategies for protecting your capital.

        Topics covered:
        • Position sizing and portfolio allocation
        • Advanced stop-loss strategies
        • Risk-adjusted return metrics
        • Correlation and diversification analysis
        
        We'll dive deep into concepts like:
        • Value at Risk (VaR)
        • Maximum drawdown management
        • Portfolio beta adjustment
        • Options hedging strategies
      `
    }
  ];

  const filteredLessons = lessons.filter(lesson => {
    const difficultyMatch = selectedDifficulty === 'all' || lesson.difficulty === selectedDifficulty;
    const topicMatch = selectedTopic === 'all' || lesson.topic === selectedTopic;
    return difficultyMatch && topicMatch;
  });

  return (
    <div className="max-w-6xl mx-auto p-6">
      <div className="mb-8">
        <h1 className="text-3xl font-bold mb-4">Trading Academy</h1>
        <div className="flex flex-wrap gap-4 mb-6">
          <div>
            <h3 className="text-sm font-medium mb-2">Difficulty</h3>
            <div className="flex gap-2">
              <Badge 
                className={`cursor-pointer ${selectedDifficulty === 'all' ? 'bg-primary' : 'bg-secondary'}`}
                onClick={() => setSelectedDifficulty('all')}
              >
                All
              </Badge>
              {difficulties.map(diff => (
                <Badge
                  key={diff}
                  className={`cursor-pointer ${selectedDifficulty === diff ? 'bg-primary' : 'bg-secondary'}`}
                  onClick={() => setSelectedDifficulty(diff)}
                >
                  {diff}
                </Badge>
              ))}
            </div>
          </div>
          <div>
            <h3 className="text-sm font-medium mb-2">Topics</h3>
            <div className="flex gap-2 flex-wrap">
              <Badge 
                className={`cursor-pointer ${selectedTopic === 'all' ? 'bg-primary' : 'bg-secondary'}`}
                onClick={() => setSelectedTopic('all')}
              >
                All Topics
              </Badge>
              {topics.map(topic => (
                <Badge
                  key={topic}
                  className={`cursor-pointer ${selectedTopic === topic ? 'bg-primary' : 'bg-secondary'}`}
                  onClick={() => setSelectedTopic(topic)}
                >
                  {topic}
                </Badge>
              ))}
            </div>
          </div>
        </div>
      </div>

      <div className="grid gap-6 md:grid-cols-2 lg:grid-cols-3">
        {filteredLessons.map((lesson, index) => (
          <Card key={index} className="hover:shadow-lg transition-shadow">
            <CardHeader>
              <div className="flex justify-between items-start">
                <CardTitle className="text-xl">{lesson.title}</CardTitle>
                <Badge variant={
                  lesson.difficulty === 'Beginner' ? 'default' :
                  lesson.difficulty === 'Intermediate' ? 'secondary' : 'destructive'
                }>
                  {lesson.difficulty}
                </Badge>
              </div>
              <CardDescription>{lesson.description}</CardDescription>
            </CardHeader>
            <CardContent>
              <div className="space-y-4">
                <div className="flex items-center gap-4 text-sm text-muted-foreground">
                  <div className="flex items-center gap-1">
                    <ScrollText className="w-4 h-4" />
                    <span>{lesson.topic}</span>
                  </div>
                  <div className="flex items-center gap-1">
                    <BookOpen className="w-4 h-4" />
                    <span>{lesson.duration}</span>
                  </div>
                </div>
                <div className="flex justify-between items-center">
                  <div className="flex gap-2">
                    <Badge variant="outline" className="flex items-center gap-1">
                      <MessageCircle className="w-4 h-4" />
                      Ask AI
                    </Badge>
                    <Badge variant="outline" className="flex items-center gap-1">
                      <GraduationCap className="w-4 h-4" />
                      Practice
                    </Badge>
                  </div>
                  <ChevronRight className="w-5 h-5 text-muted-foreground" />
                </div>
              </div>
            </CardContent>
          </Card>
        ))}
      </div>
    </div>
  );
};

export default TradingLessons;

```
